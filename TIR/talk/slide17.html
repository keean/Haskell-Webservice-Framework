    selectBreed :: Breed -> Query [
       AnimalId :=: Maybe AnimalIdDomain :*:
       AnimalName :=: Maybe AnimalNameDomain :*:
       HNil]

    selectBreed b = do
       r1 <- table animalTable
       r2 <- restrict r1 (\r -> r!AnimalType `SQL.eq` toSqlType b)
       r3 <- project r2 (AnimalId .*. AnimalName .*. HNil)
       doSelect r3
